var fs = require('fs');
var os = require('os');

var wopt = {
	flag: "w",
	encoding: "utf8",
	mode: 0666
};
var stream = fs.createWriteStream('./data.txt',wopt);

for(var i=0;i<10;i++){
	for(var j=0;j<10;j++){
		stream.write( parseInt((i+j)%10,10) + ' ');
		if(j == 9){
			stream.write(os.EOL);
		}
	}
}
